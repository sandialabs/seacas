249  template <typename T>
250  void communicate_element_data(T *file_data, T *ioss_data, size_t comp_count) const
251  {
252  show_progress(__func__);
253  // Transfer the file-decomposition based data in 'file_data' to
254  // the ioss-decomposition based data in 'ioss_data'
255  std::vector<T> export_data(exportElementMap.size() * comp_count);
256  std::vector<T> import_data(importElementMap.size() * comp_count);
257 
258  if (comp_count == 1) {
259  for (size_t i = 0; i < exportElementMap.size(); i++) {
260  size_t index = exportElementMap[i] - m_elementOffset;
261  export_data[i] = file_data[index];
262  }
263 
264  // Transfer all local data from file_data to ioss_data...
265  for (size_t i = 0; i < localElementMap.size(); i++) {
266  size_t index = localElementMap[i];
267  ioss_data[m_importPreLocalElemIndex + i] = file_data[index];
268  }
269 
270  // Get my imported data and send my exported data...
271  Ioss::MY_Alltoallv(export_data, exportElementCount, exportElementIndex, import_data,
272  importElementCount, importElementIndex, m_comm);
273  show_progress("\tCommunication 1a finished");
274 
275  // Copy the imported data into ioss_data...
276  // Some comes before the local data...
277  for (size_t i = 0; i < m_importPreLocalElemIndex; i++) {
278  ioss_data[i] = import_data[i];
279  }
280 
281  // Some comes after the local data...
282  size_t offset = m_importPreLocalElemIndex + localElementMap.size();
283  for (size_t i = 0; i < importElementMap.size() - m_importPreLocalElemIndex; i++) {
284  ioss_data[offset + i] = import_data[m_importPreLocalElemIndex + i];
285  }
286  }
287  else {
288  for (size_t i = 0; i < exportElementMap.size(); i++) {
289  size_t index = exportElementMap[i] - m_elementOffset;
290  for (size_t j = 0; j < comp_count; j++) {
291  export_data[comp_count * i + j] = file_data[comp_count * index + j];
292  }
293  }
294 
295  // Transfer all local data from file_data to ioss_data...
296  for (size_t i = 0; i < localElementMap.size(); i++) {
297  size_t index = localElementMap[i];
298  for (size_t j = 0; j < comp_count; j++) {
299  ioss_data[comp_count * (m_importPreLocalElemIndex + i) + j] =
300  file_data[comp_count * index + j];
301  }
302  }
303 
304  std::vector<INT> export_count(exportElementCount.begin(), exportElementCount.end());
305  std::vector<INT> export_disp(exportElementIndex.begin(), exportElementIndex.end());
306  std::vector<INT> import_count(importElementCount.begin(), importElementCount.end());
307  std::vector<INT> import_disp(importElementIndex.begin(), importElementIndex.end());
308 
309  for (int i = 0; i < m_processorCount; i++) {
310  export_count[i] *= comp_count;
311  export_disp[i] *= comp_count;
312  import_count[i] *= comp_count;
313  import_disp[i] *= comp_count;
314  }
315 
316  // Get my imported data and send my exported data...
317  Ioss::MY_Alltoallv(export_data, export_count, export_disp, import_data, import_count,
318  import_disp, m_comm);
319  show_progress("\tCommunication 1b finished");
320 
321  // Copy the imported data into ioss_data...
322  // Some comes before the local data...
323  for (size_t i = 0; i < m_importPreLocalElemIndex; i++) {
324  for (size_t j = 0; j < comp_count; j++) {
325  ioss_data[comp_count * i + j] = import_data[comp_count * i + j];
326  }
327  }
328 
329  // Some comes after the local data...
330  size_t offset = m_importPreLocalElemIndex + localElementMap.size();
331  for (size_t i = 0; i < importElementMap.size() - m_importPreLocalElemIndex; i++) {
332  for (size_t j = 0; j < comp_count; j++) {
333  ioss_data[comp_count * (offset + i) + j] =
334  import_data[comp_count * (m_importPreLocalElemIndex + i) + j];
335  }
336  }

341  void communicate_set_data(T *file_data, T *ioss_data, const SetDecompositionData &set,
342  size_t comp_count) const
343  {
344  show_progress(__func__);
345  std::vector<T> recv_data;
346 
347  size_t size = set.file_count() * comp_count;
348  if (size == 0)
349  return;
350 
351  if (set.setComm_ != MPI_COMM_NULL) {
352  recv_data.resize(size);
353  if (m_processor == set.root_) {
354  std::copy(file_data, file_data + size, recv_data.begin());
355  }
356  // NOTE: This broadcast uses a split communicator, so possibly
357  // not all processors participating.
358  MPI_Bcast(recv_data.data(), size, Ioss::mpi_type(T(0)), 0, set.setComm_);
359  }
360  if (comp_count == 1) {
361  if (set.root_ == m_processor) {
362  for (size_t i = 0; i < set.ioss_count(); i++) {
363  size_t index = set.entitylist_map[i];
364  ioss_data[i] = file_data[index];
365  }
366  }
367  else {
368  // Receiving data from root...
369  for (size_t i = 0; i < set.ioss_count(); i++) {
370  size_t index = set.entitylist_map[i];
371  ioss_data[i] = recv_data[index];
372  }
373  }
374  }
375  else {
376  if (set.root_ == m_processor) {
377  for (size_t i = 0; i < set.ioss_count(); i++) {
378  size_t index = set.entitylist_map[i];
379  for (size_t j = 0; j < comp_count; j++) {
380  ioss_data[comp_count * i + j] = file_data[comp_count * index + j];
381  }
382  }
383  }
384  else {
385  // Receiving data from root...
386  for (size_t i = 0; i < set.ioss_count(); i++) {
387  size_t index = set.entitylist_map[i];
388  for (size_t j = 0; j < comp_count; j++) {

396  template <typename T, typename U>
397  void communicate_block_data(T *file_data, U *ioss_data, const BlockDecompositionData &block,
398  size_t comp_count) const
399  {
400  show_progress(__func__);
401  std::vector<U> exports;
402  exports.reserve(comp_count * block.exportMap.size());
403  std::vector<U> imports(comp_count * block.importMap.size());
404 
405  if (comp_count == 1) {
406  for (int i : block.exportMap) {
407  exports.push_back(file_data[i]);
408  }
409 
410  // Get my imported data and send my exported data...
411  Ioss::MY_Alltoallv(exports, block.exportCount, block.exportIndex, imports,
412  block.importCount, block.importIndex, m_comm);
413 
414  // Map local and imported data to ioss_data.
415  for (size_t i = 0; i < block.localMap.size(); i++) {
416  ioss_data[i + block.localIossOffset] = file_data[block.localMap[i]];
417  }
418 
419  for (size_t i = 0; i < block.importMap.size(); i++) {
420  ioss_data[block.importMap[i]] = imports[i];
421  }
422  }
423  else {
424  for (int i : block.exportMap) {
425  for (size_t j = 0; j < comp_count; j++) {
426  exports.push_back(file_data[i * comp_count + j]);
427  }
428  }
429 
430  std::vector<int> export_count(block.exportCount.begin(), block.exportCount.end());
431  std::vector<int> export_disp(block.exportIndex.begin(), block.exportIndex.end());
432  std::vector<int> import_count(block.importCount.begin(), block.importCount.end());
433  std::vector<int> import_disp(block.importIndex.begin(), block.importIndex.end());
434 
435  for (int i = 0; i < m_processorCount; i++) {
436  export_count[i] *= comp_count;
437  export_disp[i] *= comp_count;
438  import_count[i] *= comp_count;
439  import_disp[i] *= comp_count;
440  }
441 
442  // Get my imported data and send my exported data...
443  Ioss::MY_Alltoallv(exports, export_count, export_disp, imports, import_count, import_disp,
444  m_comm);
445  show_progress("\tCommunication 1 finished");
446 
447  // Map local and imported data to ioss_data.
448  for (size_t i = 0; i < block.localMap.size(); i++) {
449  for (size_t j = 0; j < comp_count; j++) {
450  ioss_data[(i + block.localIossOffset) * comp_count + j] =
451  file_data[block.localMap[i] * comp_count + j];
452  }
453  }
454 
455  for (size_t i = 0; i < block.importMap.size(); i++) {
456  for (size_t j = 0; j < comp_count; j++) {
457  ioss_data[block.importMap[i] * comp_count + j] = imports[i * comp_count + j];
458  }
459  }

584  // This processor also appears to the Ioss clients to own other
585  // element and node data based on the decomposition. This is the
586  // 'ioss' data.
587  //
588  // The indices in 'local_element_map' are the elements that are
589  // common to both the 'file' data and the 'ioss' data.
590  // local_element_map[i] contains the location in 'file' data for
591  // the 'ioss' data at location 'i+import_pre_local_elem_index'
592  //
593  // local_element_map[i]+m_elementOffset is the 0-based global index
594  //
595  // The indices in 'import_element_map' map the data received via
596  // mpi communication from other processors into 'ioss' data.
597  // if 'ind=import_element_map[i]', then ioss[ind] = comm_recv[i]
598  // Note that this is the reverse direction of the
599  // local_element_map mapping.
600  //
601  // The indices in 'export_element_map' are used to pull from
602  // 'file' data into the comm_send vector. if 'ind =
603  // export_element_map[i]', then 'comm_send[i] = file[ind]' for i =
604  // 0..#exported_elements
605  //
606  // local_element_map.size() + import_element_map.size() == #
607  // ioss elements on this processor.
608  //
609  // local_element_map.size() + export_element_map.size() == #
610  // file elements on this processor.
611  //
612  // export_element_map and import_element_map are sorted.
613  // The primary key is processor order followed by global id.
614  // The processor association is via 'export_proc_disp' and
615  // 'import_proc_disp' Both are of size '#processors+1' and
616  // the elements for processor p range from [X_proc_disp[p] to
617  // X_proc_disp[p+1])
