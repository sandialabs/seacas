93  // Return number of nodes in the connection shared with the donor zone.
94  size_t get_shared_node_count() const
95  {
96  size_t snc = 1;
97  for (int i = 0; i < 3; i++) {
98  snc *= (std::abs(m_ownerRangeEnd[i] - m_ownerRangeBeg[i]) + 1);
99  }
100  return snc;
101  }
