32  /** \brief Base class for all 'grouping' entities.
33  * The following derived classes are typical:
34  *
35  * -- NodeSet -- grouping of nodes (0d topology)
36  *
37  * -- EdgeSet -- grouping of edges (1d topology)
38  *
39  * -- FaceSet -- grouping of faces (2d topology) [Surface]
40  *
41  * Similarly, there is:
42  *
43  * -- NodeBlock -- grouping of 'similar' nodes (same degree of freedom, ...)
44  *
45  * -- ElementBlock -- grouping of 'similar' elements (same element topology,
46  * attributes, ...)
47  * 0d, 1d, 2d, 3d topology possible -- e.g., sphere, bar, quad, hex
48  *
49  * A Region is also a grouping entity, except that its list of subentites
50  * are other GroupingEntities. That is, it maintains a list of NodeBlocks,
51  * ElementBlocks, NodeLists, CommLists and Surfaces. [Similar to the
52  * "Composite Patter" in Design Patterns] All interface to GroupingEntities
53  * is through the Region class; clients of the IO subsystem have no direct
54  * access to the underlying GroupingEntities (other than the Region).
55  *
56  * Each GroupingEntity contains:
57  *
58  * -- name
59  *
60  * -- MeshEntities of the specified topological dimension
61  *
62  * -- Optional attributes, either global (applied to the groupingentity), or
63  * unique value(s) to be applied to each subentity.
64  *
65  * -- Data items
66  */
67  class GroupingEntity

434 /** \brief Read type 'T' field data from the database file into memory using a std::vector.
435  *
436  * \param[in] field_name The name of the field to read.
437  * \param[out] data The data.
438  * \returns The number of values read.
439  *
440  */
441 template <typename T>
442 int64_t Ioss::GroupingEntity::get_field_data(const std::string &field_name,
443  std::vector<T> & data) const
444 {
445  verify_field_exists(field_name, "input");
446 
447  Ioss::Field field = get_field(field_name);
448  field.check_type(Ioss::Field::get_field_type(T(0)));
449 
450  data.resize(field.raw_count() * field.raw_storage()->component_count());
451  size_t data_size = data.size() * sizeof(T);
452  auto retval = internal_get_field_data(field, data.data(), data_size);
453 
454  // At this point, transform the field if specified...
455  if (retval >= 0) {
456  field.transform(data.data());
457  }
458 
459  return retval;
460 }

462 /** \brief Write type 'T' field data from memory into the database file using a std::vector.
463  *
464  * \param[in] field_name The name of the field to write.
465  * \param[in] data The data.
466  * \returns The number of values written.
467  *
468  */
469 template <typename T>
470 int64_t Ioss::GroupingEntity::put_field_data(const std::string & field_name,
471  const std::vector<T> &data) const
472 {
473  verify_field_exists(field_name, "output");
474 
475  Ioss::Field field = get_field(field_name);
476  field.check_type(Ioss::Field::get_field_type(T(0)));
477  size_t data_size = data.size() * sizeof(T);
478  if (field.has_transform()) {
479  // Need non-const data since the transform will change the users data.
480  std::vector<T> nc_data(data);
481  field.transform(nc_data.data());
482  return internal_put_field_data(field, nc_data.data(), data_size);
483  }
484 
485  T *my_data = const_cast<T *>(data.data());
486  return internal_put_field_data(field, my_data, data_size);
487 }

505 /** \brief Read field data from the database file into memory using a 1-D Kokkos:::View.
506  *
507  * \tparam T The data type.
508  * \tparam Args The other template arguments for data.
509  * \param[in] field_name The name of the field to read.
510  * \param[out] data The data.
511  * \returns The number of values read.
512  *
513  */
514 template <typename T, typename... Args>
515 int64_t Ioss::GroupingEntity::get_field_data(const std::string & field_name,
516  Kokkos::View<T *, Args...> &data) const
517 {
518  typedef Kokkos::View<T *, Args...> ViewType;
519 
520  verify_field_exists(field_name, "input");
521 
522  Ioss::Field field = get_field(field_name);
523 
524  // Resize the view
525  int new_view_size = field.raw_count() * field.raw_storage()->component_count();
526  Kokkos::resize(data, new_view_size);
527  size_t data_size = new_view_size * sizeof(T);
528 
529  // Create a host mirror view. (No memory allocation if data is in HostSpace.)
530  typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
531 
532  // Extract a pointer to the underlying allocated memory of the host view.
533  T *host_data_ptr = host_data.data();
534 
535  // Extract the data from disk to the underlying memory pointed to by host_data_ptr.
536  auto retval = internal_get_field_data(field, host_data_ptr, data_size);
537 
538  // At this point, transform the field if specified...
539  if (retval >= 0)
540  field.transform(host_data_ptr);
541 
542  // Copy the data to the device. (No op if data is in HostSpace.)
543  Kokkos::deep_copy(data, host_data);
544 
545  return retval;
546 }

548 /** \brief Read field data from the database file into memory using a 2-D Kokkos:::View.
549  *
550  * \tparam T The data type
551  * \tparam Args The other template arguments for data.
552  * \param[in] field_name The name of the field to read.
553  * \param[out] data The data.
554  * \returns The number of values read.
555  *
556  */
557 template <typename T, typename... Args>
558 int64_t Ioss::GroupingEntity::get_field_data(const std::string & field_name,
559  Kokkos::View<T **, Args...> &data) const
560 {
561  typedef Kokkos::View<T **, Args...> ViewType;
562 
563  verify_field_exists(field_name, "input");
564 
565  Ioss::Field field = get_field(field_name);
566 
567  // Resize the view
568  int new_view_size_left = field.raw_count();
569  int new_view_size_right = field.raw_storage()->component_count();
570  Kokkos::resize(data, new_view_size_left, new_view_size_right);
571  size_t data_size = new_view_size_left * new_view_size_right * sizeof(T);
572 
573  // Create and allocate an array to hold the data temporarily.
574  // This is necessary to ensure the data is placed in the correct
575  // location in the 2-D array, avoiding incorrect placement due
576  // to Views with padded dimensions.
577  T *data_array = new T[data_size];
578 
579  // Create a host mirror view. (No memory allocation if data is in HostSpace.)
580  typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
581 
582  // Extract the data from disk to the underlying memory pointed to by host_data_ptr.
583  auto retval = internal_get_field_data(field, data_array, data_size);
584 
585  // At this point, transform the field if specified...
586  if (retval >= 0)
587  field.transform(data_array);
588 
589  // Copy the data to the host Mirror view.
590  // The host mirror view has the same layout as the device view.
591  // For CUDA, this will be LayoutLeft. In this case, the loop order
592  // chosen here will be slower than the reversed loop order.
593  // However, The time for this extra in-memory copy is small
594  // compared with the time to copy from disk into memory.
595  for (int i = 0; i < new_view_size_left; ++i) {
596  for (int j = 0; j < new_view_size_right; ++j) {
597  host_data(i, j) = data_array[new_view_size_right * i + j];
598  }
599  }
600 
601  // Delete the temporary array
602  delete[] data_array;
603 
604  // Copy the data to the device. (No op if data is in HostSpace.)
605  Kokkos::deep_copy(data, host_data);
606 
607  return retval;
608 }

646 /** \brief Write field data from memory into the database file using a 2-D Kokkos::View.
647  *
648  * \tparam T The data type
649  * \tparam Args The other template arguments for data.
650  * \param[in] field_name The name of the field to write.
651  * \param[in] data The data.
652  * \returns The number of values written.
653  *
654  */
655 template <typename T, typename... Args>
656 int64_t Ioss::GroupingEntity::put_field_data(const std::string & field_name,
657  Kokkos::View<T **, Args...> &data) const
658 {
659  typedef Kokkos::View<T **, Args...> ViewType;
660 
661  verify_field_exists(field_name, "output");
662 
663  Ioss::Field field = get_field(field_name);
664 
665  int view_size_left = data.extent(0);
666  int view_size_right = data.extent(1);
667  size_t data_size = field.raw_count() * field.raw_storage()->component_count() * sizeof(T);
668 
669  if (view_size_left * view_size_right * sizeof(T) != data_size) {
670  std::ostringstream errmsg;
671  errmsg << "\nERROR: View dimensions are inconsistent with field raw count or raw storage "
672  "component count"
673  << "for field" << field_name << "\n\n";
674  IOSS_ERROR(errmsg);
675  }
676 
677  // Create a host mirror view. (No memory allocation if data is in HostSpace.)
678  typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
679 
680  // Copy the data to the host. (No op if data is in HostSpace.)
681  Kokkos::deep_copy(host_data, data);
682 
683  // Create and allocate an array to hold the data temporarily.
684  // This is necessary to ensure the data is taken from the correct
685  // location in the 2-D array, avoiding incorrect location due
686  // to Views with padded dimensions.
687  T *data_array = new T[data_size];
688 
689  // Copy the data from the host Mirror view.
690  // The host mirror view has the same layout as the device view.
691  // For CUDA, this will be LayoutLeft. In this case, the loop order
692  // chosen here will be slower than the reversed loop order.
693  // However, The time for this extra in-memory copy is small
694  // compared with the time to copy to disk from memory.
695  for (int i = 0; i < view_size_left; ++i) {
696  for (int j = 0; j < view_size_right; ++j) {
697  data_array[view_size_right * i + j] = host_data(i, j);
698  }
699  }
700 
701  // Transform the field
702  field.transform(data_array);
703 
704  // Copy the data to disk from the underlying memory pointed to by data_array.
705  auto retval = internal_put_field_data(field, data_array, data_size);
706 
707  // Delete the temporary array
708  delete[] data_array;
709 
710  return retval;
711 }
