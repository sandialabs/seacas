135  template <typename T>
136  int MY_Alltoallv64(const std::vector<T> &sendbuf, const std::vector<int64_t> &sendcounts,
137  const std::vector<int64_t> &senddisp, std::vector<T> &recvbuf,
138  const std::vector<int64_t> &recvcounts, const std::vector<int64_t> &recvdisp,
139  MPI_Comm comm)
140  {
141  int processor_count = 0;
142  int my_processor = 0;
143  MPI_Comm_size(comm, &processor_count);
144  MPI_Comm_rank(comm, &my_processor);
145 
146  // Verify that all 'counts' can fit in an integer. Symmetric
147  // communication, so recvcounts are sendcounts on another processor.
148  for (int i = 0; i < processor_count; i++) {
149  int snd_cnt = static_cast<int>(sendcounts[i]);
150  if (static_cast<int64_t>(snd_cnt) != sendcounts[i]) {
151  std::ostringstream errmsg;
152  errmsg << "ERROR: The number of items that must be communicated via MPI calls from\n"
153  << " processor " << my_processor << " to processor " << i << " is "
154  << sendcounts[i]
155  << "\n which exceeds the storage capacity of the integers "
156  "used by MPI functions.\n";
157  IOSS_ERROR(errmsg);
158  }
159  }
160 
161  size_t pow_2 = Ioss::Utils::power_2(processor_count);
162 
163  for (size_t i = 1; i < pow_2; i++) {
164  MPI_Status status{};
165 
166  int tag = 24713;
167  size_t exchange_proc = i ^ my_processor;
168  if (exchange_proc < static_cast<size_t>(processor_count)) {
169  int snd_cnt = static_cast<int>(
170  sendcounts[exchange_proc]); // Converts from int64_t to int as needed by mpi
171  int rcv_cnt = static_cast<int>(recvcounts[exchange_proc]);
172  if (static_cast<size_t>(my_processor) < exchange_proc) {
173  MPI_Send((void *)&sendbuf[senddisp[exchange_proc]], snd_cnt, mpi_type(T(0)),
174  exchange_proc, tag, comm);
175  MPI_Recv(&recvbuf[recvdisp[exchange_proc]], rcv_cnt, mpi_type(T(0)), exchange_proc, tag,
176  comm, &status);
177  }
178  else {
179  MPI_Recv(&recvbuf[recvdisp[exchange_proc]], rcv_cnt, mpi_type(T(0)), exchange_proc, tag,
180  comm, &status);
181  MPI_Send((void *)&sendbuf[senddisp[exchange_proc]], snd_cnt, mpi_type(T(0)),
182  exchange_proc, tag, comm);
183  }
184  }
185  }
186 
187  // Take care of this processor's data movement...
188  std::copy(&sendbuf[senddisp[my_processor]],
189  &sendbuf[senddisp[my_processor] + sendcounts[my_processor]],
190  &recvbuf[recvdisp[my_processor]]);
191  return 0;
192  }

194  template <typename T>
195  int MY_Alltoallv(const std::vector<T> &sendbuf, const std::vector<int64_t> &sendcnts,
196  const std::vector<int64_t> &senddisp, std::vector<T> &recvbuf,
197  const std::vector<int64_t> &recvcnts, const std::vector<int64_t> &recvdisp,
198  MPI_Comm comm)
199  {
200 // Wrapper to handle case where send/recv counts and displacements are 64-bit integers.
201 // Two cases:
202 // 1) They are of type 64-bit integers, but only storing data in the 32-bit integer range.
203 // -- if (sendcnts[#proc-1] + senddisp[#proc-1] < 2^31, then we are ok
204 // 2) They are of type 64-bit integers, and storing data in the 64-bit integer range.
205 // -- call special alltoallv which does point-to-point sends
206 #if 1
207  int processor_count = 0;
208  MPI_Comm_size(comm, &processor_count);
209  size_t max_comm = sendcnts[processor_count - 1] + senddisp[processor_count - 1];
210  size_t one = 1;
211  if (max_comm < one << 31) {
212  // count and displacement data in range, need to copy to integer vector.
213  std::vector<int> send_cnt(sendcnts.begin(), sendcnts.end());
214  std::vector<int> send_dis(senddisp.begin(), senddisp.end());
215  std::vector<int> recv_cnt(recvcnts.begin(), recvcnts.end());
216  std::vector<int> recv_dis(recvdisp.begin(), recvdisp.end());
217  return MPI_Alltoallv((void *)sendbuf.data(), send_cnt.data(), send_dis.data(), mpi_type(T(0)),
218  (void *)recvbuf.data(), recv_cnt.data(), recv_dis.data(), mpi_type(T(0)),
219  comm);
220  }
221  else {
222 #endif
223  // Same as if each processor sent a message to every other process with:
224  // MPI_Send(sendbuf+senddisp[i]*sizeof(sendtype),sendcnts[i], sendtype, i, tag, comm);
225  // And received a message from each processor with a call to:
226  // MPI_Recv(recvbuf+recvdisp[i]*sizeof(recvtype),recvcnts[i], recvtype, i, tag, comm);
227  return MY_Alltoallv64(sendbuf, sendcnts, senddisp, recvbuf, recvcnts, recvdisp, comm);
228 #if 1
229  }
230 #endif
231  }

246  template <typename T>
247  void ParallelUtils::global_array_minmax(std::vector<T> &local_minmax, MinMax which) const
248  {
249  PAR_UNUSED(local_minmax);
250  PAR_UNUSED(which);
251 #ifdef SEACAS_HAVE_MPI
252  if (parallel_size() > 1 && !local_minmax.empty()) {
253  if (Ioss::SerializeIO::isEnabled() && Ioss::SerializeIO::inBarrier()) {
254  std::ostringstream errmsg;
255  errmsg << "Attempting mpi while in barrier owned by " << Ioss::SerializeIO::getOwner();
256  IOSS_ERROR(errmsg);
257  }
258 
259  std::vector<T> maxout(local_minmax.size());
260  MPI_Op oper = MPI_MAX;
261  if (which == Ioss::ParallelUtils::DO_MAX) {
262  oper = MPI_MAX;
263  }
264  else if (which == Ioss::ParallelUtils::DO_MIN) {
265  oper = MPI_MIN;
266  }
267  else if (which == Ioss::ParallelUtils::DO_SUM) {
268  oper = MPI_SUM;
269  }
270 
271  const int success =
272  MPI_Allreduce((void *)(local_minmax.data()), maxout.data(),
273  static_cast<int>(local_minmax.size()), mpi_type(T()), oper, communicator_);
274  if (success != MPI_SUCCESS) {
275  std::ostringstream errmsg;
276  errmsg << "Ioss::ParallelUtils::global_array_minmax - MPI_Allreduce failed";
277  IOSS_ERROR(errmsg);
278  }
279  // Now copy back into passed in array...
280  for (size_t i = 0; i < local_minmax.size(); i++) {
281  local_minmax[i] = maxout[i];
282  }
283  }
284 #endif
285  }
