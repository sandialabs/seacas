35  // SEE: http://lemire.me/blog/2017/04/10/removing-duplicates-from-lists-quickly
36  template <typename T> size_t unique(std::vector<T> &out, bool skip_first)
37  {
38  if (out.empty())
39  return 0;
40  size_t i = 1;
41  size_t pos = 1;
42  T oldv = out[0];
43  if (skip_first) {
44  i = 2;
45  pos = 2;
46  oldv = out[1];
47  }
48  for (; i < out.size(); ++i) {
49  T newv = out[i];
50  out[pos] = newv;
51  pos += (newv != oldv);
52  oldv = newv;
53  }
54  return pos;
55  }

119  template <typename T> static void uniquify(std::vector<T> &vec, bool skip_first = false)
120  {
121  auto it = vec.begin();
122  if (skip_first) {
123  it++;
124  }
125  std::sort(it, vec.end());
126  vec.resize(unique(vec, skip_first));
127  vec.shrink_to_fit();
128  }

130  template <typename T> static void generate_index(std::vector<T> &index)
131  {
132  T sum = 0;
133  for (size_t i = 0; i < index.size() - 1; i++) {
134  T cnt = index[i];
135  index[i] = sum;
136  sum += cnt;
137  }
138  index.back() = sum;

141  template <typename T> static T find_index_location(T node, const std::vector<T> &index)
142  {
143  // 0-based node numbering
144  // index[p] = first node (0-based) on processor p
145 
146 #if 1
147  // Assume data coherence. I.e., a new search will be close to the
148  // previous search.
149  static size_t prev = 1;
150 
151  size_t nproc = index.size();
152  if (prev < nproc && index[prev - 1] <= node && index[prev] > node) {
153  return prev - 1;
154  }
155 
156  for (size_t p = 1; p < nproc; p++) {
157  if (index[p] > node) {
158  prev = p;
159  return p - 1;
160  }
161  }
162  std::ostringstream errmsg;
163  errmsg << "FATAL ERROR: find_index_location. Searching for " << node << " in:\n";
164  for (auto idx : index) {
165  errmsg << idx << ", ";
166  }
167  errmsg << "\n";
168  IOSS_ERROR(errmsg);
169 #else
170  return std::distance(index.begin(), std::upper_bound(index.begin(), index.end(), node)) - 1;
171 #endif
172  }

199  /**
200  * Returns the number of digits required to print the number.
201  * If `use_commas` is specified, then the width will be adjusted
202  * to account for the comma used every 3 digits.
203  * (1,234,567,890 would return 13)
204  * Typically used with the `fmt::print()` functions as:
205  * ```
206  * fmt::print("{:{}n}", number, number_width(number,true))
207  * fmt::print("{:{}d}", number, number_width(number,false))
208  * ```
209  */
210  inline static int number_width(const size_t number, bool use_commas = false)
211  {
212  if (number == 0) {
213  return 1;
214  }
215  int width = int(std::floor(std::log10(number))) + 1;
216  if (use_commas) {
217  width += ((width - 1) / 3);
218  }
219  return width;
220  }

222  inline static int power_2(int count)
223  {
224  // Return the power of two which is equal to or greater than `count`
225  // count = 15 -> returns 16
226  // count = 16 -> returns 16
227  // count = 17 -> returns 32
228 
229  // Use brute force...
230  int pow2 = 1;
231  while (pow2 < count) {
232  pow2 *= 2;
233  }
234  return pow2;
235  }
