97  class dispatcher
98  {
99  private:
100  int error_code_;

107  // Handle the result of XSI-compliant version of strerror_r.
108  int handle(int result)
109  {
110  // glibc versions before 2.13 return result in errno.
111  return result == -1 ? errno : result;
112  }
113 
114  // Handle the result of GNU-specific version of strerror_r.
115  int handle(char *message)
116  {
117  // If the buffer is full then the message is probably truncated.
118  if (message == buffer_ && strlen(buffer_) == buffer_size_ - 1)
119  return ERANGE;
120  buffer_ = message;
121  return 0;
122  }
123 
124  // Handle the case when strerror_r is not available.
125  int handle(internal::null<>)
126  {
127  return fallback(strerror_s(buffer_, buffer_size_, error_code_));
128  }
129 
130  // Fallback to strerror_s when strerror_r is not available.
131  int fallback(int result)
132  {
133  // If the buffer is full then the message is probably truncated.
134  return result == 0 && strlen(buffer_) == buffer_size_ - 1 ? ERANGE : result;
135  }
136 
137 #if !FMT_MSC_VER
138  // Fallback to strerror if strerror_r and strerror_s are not available.
139  int fallback(internal::null<>)
140  {
141  errno = 0;
142  buffer_ = strerror(error_code_);
143  return errno;
144  }
145 #endif

150  {
151  }

158  FMT_FUNC void format_error_code(internal::buffer<char> &out, int error_code,
159  string_view message) FMT_NOEXCEPT
160  {
161  // Report error code making sure that the output fits into
162  // inline_buffer_size to avoid dynamic memory allocation and potential
163  // bad_alloc.
164  out.resize(0);
165  static const char SEP[] = ": ";
166  static const char ERROR_STR[] = "error ";
167  // Subtract 2 to account for terminating null characters in SEP and ERROR_STR.
168  std::size_t error_code_size = sizeof(SEP) + sizeof(ERROR_STR) - 2;
169  auto abs_value = static_cast<uint32_or_64_t<int>>(error_code);
170  if (internal::is_negative(error_code)) {
171  abs_value = 0 - abs_value;
172  ++error_code_size;
173  }
174  error_code_size += internal::to_unsigned(internal::count_digits(abs_value));
175  internal::writer w(out);
176  if (message.size() <= inline_buffer_size - error_code_size) {
177  w.write(message);
178  w.write(SEP);
179  }
180  w.write(ERROR_STR);
181  w.write(error_code);
182  assert(out.size() <= inline_buffer_size);
183  }

250  {
251  // Assume little endian; pointer formatting is implementation-defined anyway.

253  while (i > 0 && n.value[i] == 0)

256  return i >= 0 ? i * char_digits + count_digits<4, unsigned>(n.value[i]) : 1;

347  // A handmade floating-point number f * pow(2, e).
348  class fp
349  {
350  private:
351  using significand_type = uint64_t;
352 
353  // All sizes are in bits.
354  // Subtract 1 to account for an implicit most significant bit in the
355  // normalized form.
356  static FMT_CONSTEXPR_DECL const int double_significand_size =
357  std::numeric_limits<double>::digits - 1;
358  static FMT_CONSTEXPR_DECL const uint64_t implicit_bit = 1ull << double_significand_size;
359 
360  public:
361  significand_type f;
362  int e;
363 
364  static FMT_CONSTEXPR_DECL const int significand_size = bits<significand_type>::value;
365 
366  fp() : f(0), e(0) {}
367  fp(uint64_t f_val, int e_val) : f(f_val), e(e_val) {}
368 
369  // Constructs fp from an IEEE754 double. It is a template to prevent compile
370  // errors on platforms where double is not IEEE754.
371  template <typename Double> explicit fp(Double d)
372  {
373  // Assume double is in the format [sign][exponent][significand].
374  using limits = std::numeric_limits<Double>;
375  const int exponent_size = bits<Double>::value - double_significand_size - 1; // -1 for sign
376  const uint64_t significand_mask = implicit_bit - 1;
377  const uint64_t exponent_mask = (~0ull >> 1) & ~significand_mask;
378  const int exponent_bias = (1 << exponent_size) - limits::max_exponent - 1;
379  auto u = bit_cast<uint64_t>(d);
380  auto biased_e = (u & exponent_mask) >> double_significand_size;
381  f = u & significand_mask;
382  if (biased_e != 0)
383  f += implicit_bit;
384  else
385  biased_e = 1; // Subnormals use biased exponent 1 (min exponent).
386  e = static_cast<int>(biased_e - exponent_bias - double_significand_size);
387  }
388 
389  // Normalizes the value converted from double and multiplied by (1 << SHIFT).
390  template <int SHIFT = 0> void normalize()
391  {
392  // Handle subnormals.
393  auto shifted_implicit_bit = implicit_bit << SHIFT;
394  while ((f & shifted_implicit_bit) == 0) {
395  f <<= 1;
396  --e;
397  }
398  // Subtract 1 to account for hidden bit.
399  auto offset = significand_size - double_significand_size - SHIFT - 1;
400  f <<= offset;
401  e -= offset;
402  }
403 
404  // Compute lower and upper boundaries (m^- and m^+ in the Grisu paper), where
405  // a boundary is a value half way between the number and its predecessor
406  // (lower) or successor (upper). The upper boundary is normalized and lower
407  // has the same exponent but may be not normalized.
408  void compute_boundaries(fp &lower, fp &upper) const
409  {
410  lower = f == implicit_bit ? fp((f << 2) - 1, e - 2) : fp((f << 1) - 1, e - 1);
411  upper = fp((f << 1) + 1, e - 1);
412  upper.normalize<1>(); // 1 is to account for the exponent shift above.
413  lower.f <<= lower.e - upper.e;
414  lower.e = upper.e;
415  }
416  };

418  // Returns an fp number representing x - y. Result may not be normalized.
419  inline fp operator-(fp x, fp y)
420  {
421  FMT_ASSERT(x.f >= y.f && x.e == y.e, "invalid operands");
422  return fp(x.f - y.f, x.e);
423  }

425  // Computes an fp number r with r.f = x.f * y.f / pow(2, 64) rounded to nearest
426  // with half-up tie breaking, r.e = x.e + y.e + 64. Result may not be
427  // normalized.
428  FMT_FUNC fp operator*(fp x, fp y)
429  {
430  int exp = x.e + y.e + 64;
431 #if FMT_USE_INT128
432  auto product = static_cast<__uint128_t>(x.f) * y.f;
433  auto f = static_cast<uint64_t>(product >> 64);
434  if ((static_cast<uint64_t>(product) & (1ULL << 63)) != 0)
435  ++f;
436  return fp(f, exp);
437 #else
438  // Multiply 32-bit parts of significands.
439  uint64_t mask = (1ULL << 32) - 1;
440  uint64_t a = x.f >> 32, b = x.f & mask;
441  uint64_t c = y.f >> 32, d = y.f & mask;
442  uint64_t ac = a * c, bc = b * c, ad = a * d, bd = b * d;
443  // Compute mid 64-bit of result and round.
444  uint64_t mid = (bd >> 32) + (ad & mask) + (bc & mask) + (1U << 31);
445  return fp(ac + (ad >> 32) + (bc >> 32) + (mid >> 32), exp);
446 #endif
447  }

449  // Returns cached power (of 10) c_k = c_k.f * pow(2, c_k.e) such that its
450  // (binary) exponent satisfies min_exponent <= c_k.e <= min_exponent + 28.
451  FMT_FUNC fp get_cached_power(int min_exponent, int &pow10_exponent)
452  {
453  const double one_over_log2_10 = 0.30102999566398114; // 1 / log2(10)
454  int index =
455  static_cast<int>(std::ceil((min_exponent + fp::significand_size - 1) * one_over_log2_10));
456  // Decimal exponent of the first (smallest) cached power of 10.
457  const int first_dec_exp = -348;
458  // Difference between 2 consecutive decimal exponents in cached powers of 10.
459  const int dec_exp_step = 8;
460  index = (index - first_dec_exp - 1) / dec_exp_step + 1;
461  pow10_exponent = first_dec_exp + index * dec_exp_step;
462  return fp(data::pow10_significands[index], data::pow10_exponents[index]);
463  }

465  enum round_direction { unknown, up, down };
466 
467  // Given the divisor (normally a power of 10), the remainder = v % divisor for
468  // some number v and the error, returns whether v should be rounded up, down, or
469  // whether the rounding direction can't be determined due to error.
470  // error should be less than divisor / 2.
471  inline round_direction get_round_direction(uint64_t divisor, uint64_t remainder, uint64_t error)
472  {
473  FMT_ASSERT(remainder < divisor, ""); // divisor - remainder won't overflow.
474  FMT_ASSERT(error < divisor, ""); // divisor - error won't overflow.
475  FMT_ASSERT(error < divisor - error, ""); // error * 2 won't overflow.
476  // Round down if (remainder + error) * 2 <= divisor.
477  if (remainder <= divisor - remainder && error * 2 <= divisor - remainder * 2)
478  return down;
479  // Round up if (remainder - error) * 2 >= divisor.
480  if (remainder >= error && remainder - error >= divisor - (remainder - error)) {
481  return up;
482  }
483  return unknown;
484  }

494  // Generates output using the Grisu digit-gen algorithm.
495  // error: the size of the region (lower, upper) outside of which numbers
496  // definitely do not round to value (Delta in Grisu3).
497  template <typename Handler>
498  digits::result grisu_gen_digits(fp value, uint64_t error, int &exp, Handler &handler)
499  {
500  fp one(1ull << -value.e, value.e);
501  // The integral part of scaled value (p1 in Grisu) = value / one. It cannot be
502  // zero because it contains a product of two 64-bit numbers with MSB set (due
503  // to normalization) - 1, shifted right by at most 60 bits.
504  uint32_t integral = static_cast<uint32_t>(value.f >> -one.e);
505  FMT_ASSERT(integral != 0, "");
506  FMT_ASSERT(integral == value.f >> -one.e, "");
507  // The fractional part of scaled value (p2 in Grisu) c = value % one.
508  uint64_t fractional = value.f & (one.f - 1);
509  exp = count_digits(integral); // kappa in Grisu.
510  // Divide by 10 to prevent overflow.
511  auto result =
512  handler.on_start(data::powers_of_10_64[exp - 1] << -one.e, value.f / 10, error * 10, exp);
513  if (result != digits::more)
514  return result;
515  // Generate digits for the integral part. This can produce up to 10 digits.
516  do {
517  uint32_t digit = 0;
518  // This optimization by miloyip reduces the number of integer divisions by
519  // one per iteration.
520  switch (exp) {
521  case 10:
522  digit = integral / 1000000000;
523  integral %= 1000000000;
524  break;
525  case 9:
526  digit = integral / 100000000;
527  integral %= 100000000;
528  break;
529  case 8:
530  digit = integral / 10000000;
531  integral %= 10000000;
532  break;
533  case 7:
534  digit = integral / 1000000;
535  integral %= 1000000;
536  break;
537  case 6:
538  digit = integral / 100000;
539  integral %= 100000;
540  break;
541  case 5:
542  digit = integral / 10000;
543  integral %= 10000;
544  break;
545  case 4:
546  digit = integral / 1000;
547  integral %= 1000;
548  break;
549  case 3:
550  digit = integral / 100;
551  integral %= 100;
552  break;
553  case 2:
554  digit = integral / 10;
555  integral %= 10;
556  break;
557  case 1:
558  digit = integral;
559  integral = 0;
560  break;
561  default: FMT_ASSERT(false, "invalid number of digits");
562  }
563  --exp;
564  uint64_t remainder = (static_cast<uint64_t>(integral) << -one.e) + fractional;
565  result = handler.on_digit(static_cast<char>('0' + digit),
566  data::powers_of_10_64[exp] << -one.e, remainder, error, exp, true);
567  if (result != digits::more)
568  return result;
569  } while (exp > 0);
570  // Generate digits for the fractional part.
571  for (;;) {
572  fractional *= 10;
573  error *= 10;
574  char digit = static_cast<char>('0' + static_cast<char>(fractional >> -one.e));
575  fractional &= one.f - 1;
576  --exp;
577  result = handler.on_digit(digit, one.f, fractional, error, exp, false);
578  if (result != digits::more)
579  return result;
580  }
581  }

583  // The fixed precision digit handler.
584  struct fixed_handler
585  {
586  char *buf;
587  int size;
588  int precision;
589  int exp10;
590  bool fixed;
591 
592  digits::result on_start(uint64_t divisor, uint64_t remainder, uint64_t error, int &exp)
593  {
594  // Non-fixed formats require at least one digit and no precision adjustment.
595  if (!fixed)
596  return digits::more;
597  // Adjust fixed precision by exponent because it is relative to decimal
598  // point.
599  precision += exp + exp10;
600  // Check if precision is satisfied just by leading zeros, e.g.
601  // format("{:.2f}", 0.001) gives "0.00" without generating any digits.
602  if (precision > 0)
603  return digits::more;
604  if (precision < 0)
605  return digits::done;
606  auto dir = get_round_direction(divisor, remainder, error);
607  if (dir == unknown)
608  return digits::error;
609  buf[size++] = dir == up ? '1' : '0';
610  return digits::done;
611  }
612 
613  digits::result on_digit(char digit, uint64_t divisor, uint64_t remainder, uint64_t error, int,
614  bool integral)
615  {
616  FMT_ASSERT(remainder < divisor, "");
617  buf[size++] = digit;
618  if (size < precision)
619  return digits::more;
620  if (!integral) {
621  // Check if error * 2 < divisor with overflow prevention.
622  // The check is not needed for the integral part because error = 1
623  // and divisor > (1 << 32) there.
624  if (error >= divisor || error >= divisor - error)
625  return digits::error;
626  }
627  else {
628  FMT_ASSERT(error == 1 && divisor > 2, "");
629  }
630  auto dir = get_round_direction(divisor, remainder, error);
631  if (dir != up)
632  return dir == down ? digits::done : digits::error;
633  ++buf[size - 1];
634  for (int i = size - 1; i > 0 && buf[i] > '9'; --i) {
635  buf[i] = '0';
636  ++buf[i - 1];
637  }
638  if (buf[0] > '9') {
639  buf[0] = '1';
640  buf[size++] = '0';
641  }
642  return digits::done;
643  }
644  };

646  // The shortest representation digit handler.
647  template <int GRISU_VERSION> struct grisu_shortest_handler
648  {
649  char *buf;
650  int size;
651  // Distance between scaled value and upper bound (wp_W in Grisu3).
652  uint64_t diff;
653 
654  digits::result on_start(uint64_t, uint64_t, uint64_t, int &) { return digits::more; }
655 
656  // Decrement the generated number approaching value from above.
657  void round(uint64_t d, uint64_t divisor, uint64_t &remainder, uint64_t error)
658  {
659  while (remainder < d && error - remainder >= divisor &&
660  (remainder + divisor < d || d - remainder >= remainder + divisor - d)) {
661  --buf[size - 1];
662  remainder += divisor;
663  }
664  }
665 
666  // Implements Grisu's round_weed.
667  digits::result on_digit(char digit, uint64_t divisor, uint64_t remainder, uint64_t error,
668  int exp, bool integral)
669  {
670  buf[size++] = digit;
671  if (remainder >= error)
672  return digits::more;
673  if (GRISU_VERSION != 3) {
674  uint64_t d = integral ? diff : diff * data::powers_of_10_64[-exp];
675  round(d, divisor, remainder, error);
676  return digits::done;
677  }
678  uint64_t unit = integral ? 1 : data::powers_of_10_64[-exp];
679  uint64_t up = (diff - 1) * unit; // wp_Wup
680  round(up, divisor, remainder, error);
681  uint64_t down = (diff + 1) * unit; // wp_Wdown
682  if (remainder < down && error - remainder >= divisor &&
683  (remainder + divisor < down || down - remainder > remainder + divisor - down)) {
684  return digits::error;
685  }
686  return 2 * unit <= remainder && remainder <= error - 4 * unit ? digits::done : digits::error;
687  }
688  };

690  template <typename Double, enable_if_t<(sizeof(Double) == sizeof(uint64_t)), int>>
691  FMT_API bool grisu_format(Double value, buffer<char> &buf, int precision, unsigned options,
692  int &exp)
693  {
694  FMT_ASSERT(value >= 0, "value is negative");
695  bool fixed = (options & grisu_options::fixed) != 0;
696  if (value <= 0) { // <= instead of == to silence a warning.
697  if (precision <= 0 || !fixed) {
698  exp = 0;
699  buf.push_back('0');
700  }
701  else {
702  exp = -precision;
703  buf.resize(precision);
704  std::uninitialized_fill_n(buf.data(), precision, '0');
705  }
706  return true;
707  }
708 
709  fp fp_value(value);
710  const int min_exp = -60; // alpha in Grisu.
711  int cached_exp10 = 0; // K in Grisu.
712  if (precision != -1) {
713  if (precision > 17)
714  return false;
715  fp_value.normalize();
716  auto cached_pow =
717  get_cached_power(min_exp - (fp_value.e + fp::significand_size), cached_exp10);
718  fp_value = fp_value * cached_pow;
719  fixed_handler handler{buf.data(), 0, precision, -cached_exp10, fixed};
720  if (grisu_gen_digits(fp_value, 1, exp, handler) == digits::error)
721  return false;
722  buf.resize(to_unsigned(handler.size));
723  }
724  else {
725  fp lower, upper; // w^- and w^+ in the Grisu paper.
726  fp_value.compute_boundaries(lower, upper);
727  // Find a cached power of 10 such that multiplying upper by it will bring
728  // the exponent in the range [min_exp, -32].
729  auto cached_pow = get_cached_power( // \tilde{c}_{-k} in Grisu.
730  min_exp - (upper.e + fp::significand_size), cached_exp10);
731  fp_value.normalize();
732  fp_value = fp_value * cached_pow;
733  lower = lower * cached_pow; // \tilde{M}^- in Grisu.
734  upper = upper * cached_pow; // \tilde{M}^+ in Grisu.
735  assert(min_exp <= upper.e && upper.e <= -32);
736  auto result = digits::result();
737  int size = 0;
738  if ((options & grisu_options::grisu3) != 0) {
739  --lower.f; // \tilde{M}^- - 1 ulp -> M^-_{\downarrow}.
740  ++upper.f; // \tilde{M}^+ + 1 ulp -> M^+_{\uparrow}.
741  // Numbers outside of (lower, upper) definitely do not round to value.
742  grisu_shortest_handler<3> handler{buf.data(), 0, (upper - fp_value).f};
743  result = grisu_gen_digits(upper, upper.f - lower.f, exp, handler);
744  size = handler.size;
745  }
746  else {
747  ++lower.f; // \tilde{M}^- + 1 ulp -> M^-_{\uparrow}.
748  --upper.f; // \tilde{M}^+ - 1 ulp -> M^+_{\downarrow}.
749  grisu_shortest_handler<2> handler{buf.data(), 0, (upper - fp_value).f};
750  result = grisu_gen_digits(upper, upper.f - lower.f, exp, handler);
751  size = handler.size;
752  }
753  if (result == digits::error)
754  return false;
755  buf.resize(to_unsigned(size));
756  }
757  exp -= cached_exp10;
758  return true;
759  }

761  template <typename Double>
762  char *sprintf_format(Double value, internal::buffer<char> &buf, sprintf_specs specs)
763  {
764  // Buffer capacity must be non-zero, otherwise MSVC's vsnprintf_s will fail.
765  FMT_ASSERT(buf.capacity() != 0, "empty buffer");
766 
767  // Build format string.
768  enum { max_format_size = 10 }; // longest format: %#-*.*Lg
769  char format[max_format_size];
770  char *format_ptr = format;
771  *format_ptr++ = '%';
772  if (specs.alt || !specs.type)
773  *format_ptr++ = '#';
774  if (specs.precision >= 0) {
775  *format_ptr++ = '.';
776  *format_ptr++ = '*';
777  }
778  if (std::is_same<Double, long double>::value)
779  *format_ptr++ = 'L';
780 
781  char type = specs.type;
782 
783  if (type == '%')
784  type = 'f';
785  else if (type == 0 || type == 'n')
786  type = 'g';
787 #if FMT_MSC_VER
788  if (type == 'F') {
789  // MSVC's printf doesn't support 'F'.
790  type = 'f';
791  }
792 #endif
793  *format_ptr++ = type;
794  *format_ptr = '\0';
795 
796  // Format using snprintf.
797  char *start = nullptr;
798  char *decimal_point_pos = nullptr;
799  for (;;) {
800  std::size_t buffer_size = buf.capacity();
801  start = &buf[0];
802  int result = format_float(start, buffer_size, format, specs.precision, value);
803  if (result >= 0) {
804  unsigned n = internal::to_unsigned(result);
805  if (n < buf.capacity()) {
806  // Find the decimal point.
807  auto p = buf.data(), end = p + n;
808  if (*p == '+' || *p == '-')
809  ++p;
810  if (specs.type != 'a' && specs.type != 'A') {
811  while (p < end && *p >= '0' && *p <= '9')
812  ++p;
813  if (p < end && *p != 'e' && *p != 'E') {
814  decimal_point_pos = p;
815  if (!specs.type) {
816  // Keep only one trailing zero after the decimal point.
817  ++p;
818  if (*p == '0')
819  ++p;
820  while (p != end && *p >= '1' && *p <= '9')
821  ++p;
822  char *where = p;
823  while (p != end && *p == '0')
824  ++p;
825  if (p == end || *p < '0' || *p > '9') {
826  if (p != end)
827  std::memmove(where, p, to_unsigned(end - p));
828  n -= static_cast<unsigned>(p - where);
829  }
830  }
831  }
832  }
833  buf.resize(n);
834  break; // The buffer is large enough - continue with formatting.
835  }
836  buf.reserve(n + 1);
837  }
838  else {
839  // If result is negative we ask to increase the capacity by at least 1,
840  // but as std::vector, the buffer grows exponentially.
841  buf.reserve(buf.capacity() + 1);
842  }
843  }
844  return decimal_point_pos;
845  }
