86  template <typename INT>
87  static void map_cgns_connectivity(const Ioss::ElementTopology *topo, size_t element_count,
88  INT *idata)
89  {
90  // Map from CGNS to IOSS/Exodus/Patran order...
91  switch (topo->shape()) {
92  case Ioss::ElementShape::HEX:
93  switch (topo->number_nodes()) {
94  case 8:
95  case 20: break;
96  case 27: {
97  // nodes 1 through 20 are the same...
98  //
99  // ioss: 21, 22, 23, 24, 25, 26, 27 [zero-based: 20, 21, 22, 23, 24, 25, 26]
100  // cgns: 27, 21, 26, 25, 23, 22, 24 [zero-based: 26, 20, 25, 24, 22, 21, 23]
101  static std::array<int, 7> hex27_map{26, 20, 25, 24, 22, 21, 23};
102  for (size_t i = 0; i < element_count; i++) {
103  size_t con_beg = 27 * i; // start of connectivity for i'th element.
104  std::array<int, 7> reorder;
105  for (size_t j = 0; j < 7; j++) {
106  reorder[j] = idata[con_beg + hex27_map[j]];
107  }
108 
109  for (size_t j = 0; j < 7; j++) {
110  idata[con_beg + 20 + j] = reorder[j];
111  }
112  }
113  }
114  }
115  break;
116  default:
117  // do nothing cgns ordering matches ioss/exodus/patran (or not handled yet... todo: add
118  // error checking)
119  ;
120  }
121  }

160  template <typename INT>
161  static void map_cgns_face_to_ioss(const Ioss::ElementTopology *parent_topo, size_t num_to_get,
162  INT *idata)
163  {
164  // The {topo}_map[] arrays map from CGNS face# to IOSS face#.
165  // See http://cgns.github.io/CGNS_docs_current/sids/conv.html#unstructgrid
166  // NOTE: '0' for first entry is to account for 1-based face numbering.
167 
168  switch (parent_topo->shape()) {
169  case Ioss::ElementShape::HEX:
170  static int hex_map[] = {0, 5, 1, 2, 3, 4, 6};
171  for (size_t i = 0; i < num_to_get; i++) {
172  idata[2 * i + 1] = hex_map[idata[2 * i + 1]];
173  }
174  break;
175 
176  case Ioss::ElementShape::TET:
177  static int tet_map[] = {0, 4, 1, 2, 3};
178  for (size_t i = 0; i < num_to_get; i++) {
179  idata[2 * i + 1] = tet_map[idata[2 * i + 1]];
180  }
181  break;
182 
183  case Ioss::ElementShape::PYRAMID:
184  static int pyr_map[] = {0, 5, 1, 2, 3, 4};
185  for (size_t i = 0; i < num_to_get; i++) {
186  idata[2 * i + 1] = pyr_map[idata[2 * i + 1]];
187  }
188  break;
189 
190  case Ioss::ElementShape::WEDGE:
191 #if 0
192  static int wed_map[] = {0, 1, 2, 3, 4, 5}; // Same
193  // Not needed -- maps 1 to 1
194  for (size_t i=0; i < num_to_get; i++) {
195  idata[2*i+1] = wed_map[idata[2*i+1]];
196  }
197 #endif
198  break;
199  default:;
200  }
201  }

203  static void map_ioss_face_to_cgns(const Ioss::ElementTopology *parent_topo, size_t num_to_get,
204  CGNSIntVector &data)
205  {
206  // The {topo}_map[] arrays map from CGNS face# to IOSS face#.
207  // See http://cgns.github.io/CGNS_docs_current/sids/conv.html#unstructgrid
208  // NOTE: '0' for first entry is to account for 1-based face numbering.
209 
210  switch (parent_topo->shape()) {
211  case Ioss::ElementShape::HEX:
212  static int hex_map[] = {0, 2, 3, 4, 5, 1, 6};
213  for (size_t i = 0; i < num_to_get; i++) {
214  data[num_to_get * 2 + i] = hex_map[data[num_to_get * 2 + i]];
215  }
216  break;
217 
218  case Ioss::ElementShape::TET:
219  static int tet_map[] = {0, 2, 3, 4, 1};
220  for (size_t i = 0; i < num_to_get; i++) {
221  data[num_to_get * 2 + i] = tet_map[data[num_to_get * 2 + i]];
222  }
223  break;
224 
225  case Ioss::ElementShape::PYRAMID:
226  static int pyr_map[] = {0, 2, 3, 4, 5, 1};
227  for (size_t i = 0; i < num_to_get; i++) {
228  data[num_to_get * 2 + i] = pyr_map[data[num_to_get * 2 + i]];
229  }
230  break;
231 
232  case Ioss::ElementShape::WEDGE:
233 #if 0
234  static int wed_map[] = {0, 1, 2, 3, 4, 5}; // Same
235  // Not needed -- maps 1 to 1
236  for (size_t i=0; i < num_to_get; i++) {
237  data[num_to_get * 2 + i] = wed_map[data[num_to_get * 2 + i]];
238  }
239 #endif
240  break;
241  default:;
242  }
