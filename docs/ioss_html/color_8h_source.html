185 // rgb is a struct for red, green and blue colors.
186 // Using the name "rgb" makes some editors show the color in a tooltip.
187 struct rgb
188 {
189  FMT_CONSTEXPR rgb() : r(0), g(0), b(0) {}
190  FMT_CONSTEXPR rgb(uint8_t r_, uint8_t g_, uint8_t b_) : r(r_), g(g_), b(b_) {}
191  FMT_CONSTEXPR rgb(uint32_t hex) : r((hex >> 16) & 0xFF), g((hex >> 8) & 0xFF), b(hex & 0xFF) {}
192  FMT_CONSTEXPR rgb(color hex)
193  : r((uint32_t(hex) >> 16) & 0xFF), g((uint32_t(hex) >> 8) & 0xFF), b(uint32_t(hex) & 0xFF)
194  {
195  }
196  uint8_t r;
197  uint8_t g;
198  uint8_t b;
199 };

203  // color is a struct of either a rgb color or a terminal color.
204  struct color_type
205  {
206  FMT_CONSTEXPR color_type() FMT_NOEXCEPT : is_rgb(), value{} {}
207  FMT_CONSTEXPR color_type(color rgb_color) FMT_NOEXCEPT : is_rgb(true), value{}
208  {
209  value.rgb_color = static_cast<uint32_t>(rgb_color);
210  }
211  FMT_CONSTEXPR color_type(rgb rgb_color) FMT_NOEXCEPT : is_rgb(true), value{}
212  {
213  value.rgb_color = (static_cast<uint32_t>(rgb_color.r) << 16) |
214  (static_cast<uint32_t>(rgb_color.g) << 8) | rgb_color.b;
215  }
216  FMT_CONSTEXPR color_type(terminal_color term_color) FMT_NOEXCEPT : is_rgb(), value{}
217  {
218  value.term_color = static_cast<uint8_t>(term_color);
219  }
220  bool is_rgb;
221  union color_union {
222  uint8_t term_color;
223  uint32_t rgb_color;
224  } value;
225  };

230 {
231 public:
232  FMT_CONSTEXPR text_style(emphasis em = emphasis()) FMT_NOEXCEPT : set_foreground_color(),
233  set_background_color(),

238  FMT_CONSTEXPR text_style &operator|=(const text_style &rhs)
239  {
240  if (!set_foreground_color) {
241  set_foreground_color = rhs.set_foreground_color;
242  foreground_color = rhs.foreground_color;
243  }
244  else if (rhs.set_foreground_color) {
245  if (!foreground_color.is_rgb || !rhs.foreground_color.is_rgb)
246  FMT_THROW(format_error("can't OR a terminal color"));
247  foreground_color.value.rgb_color |= rhs.foreground_color.value.rgb_color;
248  }
249 
250  if (!set_background_color) {
251  set_background_color = rhs.set_background_color;
252  background_color = rhs.background_color;
253  }
254  else if (rhs.set_background_color) {
255  if (!background_color.is_rgb || !rhs.background_color.is_rgb)
256  FMT_THROW(format_error("can't OR a terminal color"));
257  background_color.value.rgb_color |= rhs.background_color.value.rgb_color;
258  }
259 
260  ems = static_cast<emphasis>(static_cast<uint8_t>(ems) | static_cast<uint8_t>(rhs.ems));
261  return *this;
262  }

362  template <typename Char> struct ansi_color_escape
363  {
364  FMT_CONSTEXPR ansi_color_escape(internal::color_type text_color, const char *esc) FMT_NOEXCEPT
365  {
366  // If we have a terminal color, we need to output another escape code
367  // sequence.
368  if (!text_color.is_rgb) {
369  bool is_background = esc == internal::data::background_color;
370  uint32_t value = text_color.value.term_color;
371  // Background ASCII codes are the same as the foreground ones but with
372  // 10 more.
373  if (is_background)
374  value += 10u;
375 
376  std::size_t index = 0;
377  buffer[index++] = static_cast<Char>('\x1b');
378  buffer[index++] = static_cast<Char>('[');
379 
380  if (value >= 100u) {
381  buffer[index++] = static_cast<Char>('1');
382  value %= 100u;
383  }
384  buffer[index++] = static_cast<Char>('0' + value / 10u);
385  buffer[index++] = static_cast<Char>('0' + value % 10u);
386 
387  buffer[index++] = static_cast<Char>('m');
388  buffer[index++] = static_cast<Char>('\0');
389  return;
390  }
391 
392  for (int i = 0; i < 7; i++) {
393  buffer[i] = static_cast<Char>(esc[i]);
394  }
395  rgb color(text_color.value.rgb_color);
396  to_esc(color.r, buffer + 7, ';');
397  to_esc(color.g, buffer + 11, ';');
398  to_esc(color.b, buffer + 15, 'm');
399  buffer[19] = static_cast<Char>(0);
400  }
401  FMT_CONSTEXPR ansi_color_escape(emphasis em) FMT_NOEXCEPT
402  {
403  uint8_t em_codes[4] = {};
404  uint8_t em_bits = static_cast<uint8_t>(em);
405  if (em_bits & static_cast<uint8_t>(emphasis::bold))
406  em_codes[0] = 1;
407  if (em_bits & static_cast<uint8_t>(emphasis::italic))
408  em_codes[1] = 3;
409  if (em_bits & static_cast<uint8_t>(emphasis::underline))
410  em_codes[2] = 4;
411  if (em_bits & static_cast<uint8_t>(emphasis::strikethrough))
412  em_codes[3] = 9;
413 
414  std::size_t index = 0;
415  for (int i = 0; i < 4; ++i) {
416  if (!em_codes[i])
417  continue;
418  buffer[index++] = static_cast<Char>('\x1b');
419  buffer[index++] = static_cast<Char>('[');
420  buffer[index++] = static_cast<Char>('0' + em_codes[i]);
421  buffer[index++] = static_cast<Char>('m');
422  }
423  buffer[index++] = static_cast<Char>(0);
424  }
425  FMT_CONSTEXPR operator const Char *() const FMT_NOEXCEPT { return buffer; }
426 
427  FMT_CONSTEXPR const Char *begin() const FMT_NOEXCEPT { return buffer; }
428  FMT_CONSTEXPR const Char *end() const FMT_NOEXCEPT { return buffer + std::strlen(buffer); }
429 
430  private:
431  Char buffer[7u + 3u * 4u + 1u];
432 
433  static FMT_CONSTEXPR void to_esc(uint8_t c, Char *out, char delimiter) FMT_NOEXCEPT
434  {
435  out[0] = static_cast<Char>('0' + c / 100);
436  out[1] = static_cast<Char>('0' + c / 10 % 10);
437  out[2] = static_cast<Char>('0' + c % 10);
438  out[3] = static_cast<Char>(delimiter);
439  }
440  };

489  template <typename Char>
490  std::basic_string<Char> vformat(const text_style &ts, basic_string_view<Char> format_str,
491  basic_format_args<buffer_context<Char>> args)
492  {
493  basic_memory_buffer<Char> buffer;
494  bool has_style = false;
495  if (ts.has_emphasis()) {
496  has_style = true;
497  ansi_color_escape<Char> escape = make_emphasis<Char>(ts.get_emphasis());
498  buffer.append(escape.begin(), escape.end());
499  }
500  if (ts.has_foreground()) {
501  has_style = true;
502  ansi_color_escape<Char> escape = make_foreground_color<Char>(ts.get_foreground());
503  buffer.append(escape.begin(), escape.end());
504  }
505  if (ts.has_background()) {
506  has_style = true;
507  ansi_color_escape<Char> escape = make_background_color<Char>(ts.get_background());
508  buffer.append(escape.begin(), escape.end());
509  }
510  internal::vformat_to(buffer, format_str, args);
511  if (has_style) {
