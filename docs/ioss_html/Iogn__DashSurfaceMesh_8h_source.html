130  DashSurfaceData(const std::vector<double> &coords, const std::vector<int> &connectivity1,
131  const std::vector<int> &connectivity2)
132  : coordinates(coords), surfaceAConnectivity(connectivity1),
133  surfaceBConnectivity(connectivity2)
134  {
135  this->setSerialDefaults();
136  }
137 
138  private:
139  void setSerialDefaults()
140  {
141  globalNumberOfNodes = coordinates.size() / SPATIAL_DIMENSION;
142 
143  globalNumberOfElementsSurface1 = surfaceBConnectivity.size() / NUM_NODES_PER_QUAD_FACE;
144  globalNumberOfElementsSurface2 = surfaceAConnectivity.size() / NUM_NODES_PER_QUAD_FACE;
145  globalNumberOfElements = globalNumberOfElementsSurface1 + globalNumberOfElementsSurface2;
146 
147  globalIdsOfLocalElements.resize(globalNumberOfElements);
148  globalIdsOfLocalNodes.resize(globalNumberOfNodes);
149 
150  for (size_t i = 0; i < globalIdsOfLocalElements.size(); i++) {
151  globalIdsOfLocalElements[i] = i + 1;
152  }
153 
154  for (size_t i = 0; i < globalIdsOfLocalNodes.size(); i++) {
155  globalIdsOfLocalNodes[i] = i + 1;
156  }
157  }
158  };
