73  struct FaceEqual
74  {
75  bool operator()(const Face &left, const Face &right) const
76  {
77  if (left.hashId_ != right.hashId_) {
78  return false;
79  }
80  // Hash (hashId_) is equal
81  // Check whether same vertices (can be in different order)
82  // Most (All?) of the time, there are no hashId_ collisions, so this test will not
83  // find a difference and the function will return 'true'
84  // However, for some reason, removing this check does not change the execution time
85  // appreiciably...
86  for (auto lvert : left.connectivity_) {
87  if (std::find(right.connectivity_.cbegin(), right.connectivity_.cend(), lvert) ==
88  right.connectivity_.cend()) {
89  // Not found, therefore not the same.
90  return false;
91  }
92  }
93  return true;
94  }
95  };
