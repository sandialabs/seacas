28  protected:
29  // The put-area is actually always empty. This makes the implementation
30  // simpler and has the advantage that the streambuf and the buffer are always
31  // in sync and sputc never writes into uninitialized memory. The obvious
32  // disadvantage is that each call to sputc always results in a (virtual) call
33  // to overflow. There is no disadvantage here for sputn since this always
34  // results in a call to xsputn.
35 
36  int_type overflow(int_type ch = traits_type::eof()) FMT_OVERRIDE
37  {
38  if (!traits_type::eq_int_type(ch, traits_type::eof()))
39  buffer_.push_back(static_cast<Char>(ch));
40  return ch;
41  }

43  std::streamsize xsputn(const Char *s, std::streamsize count) FMT_OVERRIDE
44  {
45  buffer_.append(s, s + count);
46  return count;
47  }

76  // Write the content of buf to os.
77  template <typename Char> void write(std::basic_ostream<Char> &os, buffer<Char> &buf)
78  {
79  const Char *buf_data = buf.data();
80  using unsigned_streamsize = std::make_unsigned<std::streamsize>::type;
81  unsigned_streamsize size = buf.size();
82  unsigned_streamsize max_size = to_unsigned((std::numeric_limits<std::streamsize>::max)());
83  do {
84  unsigned_streamsize n = size <= max_size ? size : max_size;
85  os.write(buf_data, static_cast<std::streamsize>(n));
86  buf_data += n;
87  size -= n;
88  } while (size != 0);
89  }

91  template <typename Char, typename T> void format_value(buffer<Char> &buf, const T &value)
92  {
93  formatbuf<Char> format_buf(buf);
94  std::basic_ostream<Char> output(&format_buf);
95  output.exceptions(std::ios_base::failbit | std::ios_base::badbit);
96  output << value;
97  buf.resize(buf.size());
98  }
