85  template <typename T, typename Context> class arg_converter
86  {
87  private:
88  using char_type = typename Context::char_type;

94  arg_converter(basic_format_arg<Context> &arg, char_type type) : arg_(arg), type_(type) {}

96  void operator()(bool value)
97  {
98  if (type_ != 's')
99  operator()<bool>(value);
100  }
101 
102  template <typename U, FMT_ENABLE_IF(std::is_integral<U>::value)> void operator()(U value)
103  {
104  bool is_signed = type_ == 'd' || type_ == 'i';
105  using target_type = conditional_t<std::is_same<T, void>::value, U, T>;
106  if (const_check(sizeof(target_type) <= sizeof(int))) {
107  // Extra casts are used to silence warnings.
108  if (is_signed) {
109  arg_ = internal::make_arg<Context>(static_cast<int>(static_cast<target_type>(value)));
110  }
111  else {
112  using unsigned_type = typename make_unsigned_or_bool<target_type>::type;
113  arg_ =
114  internal::make_arg<Context>(static_cast<unsigned>(static_cast<unsigned_type>(value)));
115  }
116  }
117  else {
118  if (is_signed) {
119  // glibc's printf doesn't sign extend arguments of smaller types:
120  // std::printf("%lld", -42); // prints "4294967254"
121  // but we don't have to do the same because it's a UB.
122  arg_ = internal::make_arg<Context>(static_cast<long long>(value));
123  }
124  else {
125  arg_ = internal::make_arg<Context>(
126  static_cast<typename make_unsigned_or_bool<U>::type>(value));
127  }
128  }
129  }
130 
131  template <typename U, FMT_ENABLE_IF(!std::is_integral<U>::value)>
132  void operator()(U) {} // No conversion needed for non-integral types.
133  };

260  template <typename T, FMT_ENABLE_IF(std::is_integral<T>::value)> iterator operator()(T value)
261  {
262  // MSVC2013 fails to compile separate overloads for bool and char_type so
263  // use std::is_same instead.
264  if (std::is_same<T, bool>::value) {
265  format_specs &fmt_specs = *this->specs();
266  if (fmt_specs.type != 's')
267  return base::operator()(value ? 1 : 0);
268  fmt_specs.type = 0;
269  this->write(value != 0);
270  }
271  else if (std::is_same<T, char_type>::value) {
272  format_specs &fmt_specs = *this->specs();
273  if (fmt_specs.type && fmt_specs.type != 'c')
274  return (*this)(static_cast<int>(value));
275  fmt_specs.sign = sign::none;
276  fmt_specs.alt = false;
277  fmt_specs.align = align::right;
278  return base::operator()(value);
279  }
280  else {
281  return base::operator()(value);
282  }
283  return this->out();
284  }

406 template <typename OutputIt, typename Char>
407 void basic_printf_context<OutputIt, Char>::parse_flags(format_specs &specs, const Char *&it,
408  const Char *end)
409 {
410  for (; it != end; ++it) {
411  switch (*it) {
412  case '-': specs.align = align::left; break;
413  case '+': specs.sign = sign::plus; break;
414  case '0': specs.fill[0] = '0'; break;
415  case ' ': specs.sign = sign::space; break;
416  case '#': specs.alt = true; break;
417  default: return;
418  }
419  }

433 template <typename OutputIt, typename Char>
434 unsigned basic_printf_context<OutputIt, Char>::parse_header(const Char *&it, const Char *end,
435  format_specs &specs)
436 {
437  unsigned arg_index = std::numeric_limits<unsigned>::max();
438  char_type c = *it;
439  if (c >= '0' && c <= '9') {
440  // Parse an argument index (if followed by '$') or a width possibly
441  // preceded with '0' flag(s).
442  internal::error_handler eh;
443  unsigned value = parse_nonnegative_int(it, end, eh);
444  if (it != end && *it == '$') { // value is an argument index
445  ++it;
446  arg_index = value;
447  }
448  else {
449  if (c == '0')
450  specs.fill[0] = '0';
451  if (value != 0) {
452  // Nonzero value means that we parsed width and don't need to
453  // parse it or flags again, so return now.
454  specs.width = value;
455  return arg_index;
456  }
457  }
458  }
459  parse_flags(specs, it, end);
460  // Parse width.
461  if (it != end) {
462  if (*it >= '0' && *it <= '9') {
463  internal::error_handler eh;
464  specs.width = parse_nonnegative_int(it, end, eh);
465  }
466  else if (*it == '*') {
467  ++it;
468  specs.width = visit_format_arg(internal::printf_width_handler<char_type>(specs), get_arg());
469  }
470  }
471  return arg_index;
472 }

474 template <typename OutputIt, typename Char>
475 template <typename ArgFormatter>
476 OutputIt basic_printf_context<OutputIt, Char>::format()
477 {
478  auto my_out = this->out();
479  const Char *start = parse_ctx_.begin();
480  const Char *end = parse_ctx_.end();
481  auto it = start;
482  while (it != end) {
483  char_type c = *it++;
484  if (c != '%')
485  continue;
486  if (it != end && *it == c) {
487  my_out = std::copy(start, it, my_out);
488  start = ++it;
489  continue;
490  }
491  my_out = std::copy(start, it - 1, my_out);
492 
493  format_specs specs;
494  specs.align = align::right;
495 
496  // Parse argument index, flags and width.
497  unsigned arg_index = parse_header(it, end, specs);
498 
499  // Parse precision.
500  if (it != end && *it == '.') {
501  ++it;
502  c = it != end ? *it : 0;
503  if ('0' <= c && c <= '9') {
504  internal::error_handler eh;
505  specs.precision = static_cast<int>(parse_nonnegative_int(it, end, eh));
506  }
507  else if (c == '*') {
508  ++it;
509  specs.precision = visit_format_arg(internal::printf_precision_handler(), get_arg());
510  }
511  else {
512  specs.precision = 0;
513  }
514  }
515 
516  format_arg my_arg = get_arg(arg_index);
517  if (specs.alt && visit_format_arg(internal::is_zero_int(), my_arg))
518  specs.alt = false;
519  if (specs.fill[0] == '0') {
520  if (my_arg.is_arithmetic())
521  specs.align = align::numeric;
522  else
523  specs.fill[0] = ' '; // Ignore '0' flag for non-numeric types.
524  }
525 
526  // Parse length and convert the argument to the required type.
527  c = it != end ? *it++ : 0;
528  char_type t = it != end ? *it : 0;
529  using internal::convert_arg;
530  switch (c) {
531  case 'h':
532  if (t == 'h') {
533  ++it;
534  t = it != end ? *it : 0;
535  convert_arg<signed char>(my_arg, t);
536  }
537  else {
538  convert_arg<short>(my_arg, t);
539  }
540  break;
541  case 'l':
542  if (t == 'l') {
543  ++it;
544  t = it != end ? *it : 0;
545  convert_arg<long long>(my_arg, t);
546  }
547  else {
548  convert_arg<long>(my_arg, t);
549  }
550  break;
551  case 'j': convert_arg<intmax_t>(my_arg, t); break;
552  case 'z': convert_arg<std::size_t>(my_arg, t); break;
553  case 't': convert_arg<std::ptrdiff_t>(my_arg, t); break;
554  case 'L':
555  // printf produces garbage when 'L' is omitted for long double, no
556  // need to do the same.
557  break;
558  default: --it; convert_arg<void>(my_arg, c);
559  }
560 
561  // Parse type.
562  if (it == end)
563  FMT_THROW(format_error("invalid format string"));
564  specs.type = static_cast<char>(*it++);
565  if (my_arg.is_integral()) {
566  // Normalize type.
567  switch (specs.type) {
568  case 'i':
569  case 'u': specs.type = 'd'; break;
570  case 'c':
571  visit_format_arg(internal::char_converter<basic_printf_context>(my_arg), my_arg);
572  break;
573  }
574  }
575 
576  start = it;
577 
578  // Format argument.
579  visit_format_arg(ArgFormatter(my_out, specs, *this), my_arg);
580  }
581  return std::copy(start, it, my_out);
582 }
