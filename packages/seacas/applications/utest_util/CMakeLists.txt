
TRIBITS_SUBPACKAGE(UTestUtil)

SET(UTESTUTIL_VERSION_MAJOR "1")
SET(UTESTUTIL_VERSION_MINOR "00")
SET(UTESTUTIL_VERSION_PATCH "0")
SET(UTESTUTIL_VERSION "${UTESTUTIL_VERSION_MAJOR}.${UTESTUTIL_VERSION_MINOR}")
SET(UTESTUTIL_VERSION_FULL "${UTESTUTIL_VERSION}.${UTESTUTIL_VERSION_PATCH}")

IF (TPL_ENABLE_MPI)
  SET(SEACAS_HAVE_MPI ON)
ENDIF()

TRIBITS_INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}"
  "${CMAKE_CURRENT_BINARY_DIR}"
  )

SET(SOURCES "")
APPEND_GLOB(SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.C)

IF (TPL_ENABLE_GTest)
TRIBITS_ADD_LIBRARY(
  utest_util
  NOINSTALLHEADERS
  SOURCES ${SOURCES}
  )
ENDIF()

TRIBITS_SUBPACKAGE_POSTPROCESS()
